

keytool -list -v -keystore keystore/kafka.keystore.jks -storepass "keystorepassword"       ─╯
# → Entry type: PrivateKeyEntry
# → Certificate chain length: 2
# → SubjectAlternativeName에 SAN들이 보이는지 확인

---

kubectl create secret generic kafka-jks \
  --from-file=kafka.truststore.jks=truststore/kafka.truststore.jks \
  --from-file=kafka.keystore.jks=keystore/kafka.keystore.jks

---

tls:
  type: JKS
  existingSecret: kafka-jks
  keystorePassword: keystorepassword
  truststorePassword: truststorepassword

listeners:
  client:
    protocol: SASL_SSL   # 또는 SSL
  interbroker:
    protocol: SSL
  controller:
    protocol: SSL
